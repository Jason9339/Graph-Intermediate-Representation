\tikzstyle{profile} = [regular polygon,regular polygon sides=6, minimum width=3cm, minimum height=1cm, text centered, draw=brown, fill=brown!30]

\begin{tikzpicture}[node distance=2cm]

\node (sawmill) [process] {Sawmill Operations};

\node (planing) [process, below right=of sawmill] {Planing};
\node (waste) [material, below=of sawing] {Wood Waste};
\node (emissions) [emission, below=of waste] {Particulate Emissions};

\draw [arrow] (sawmill) -- node[anchor=east] {includes} (raw);
\draw [arrow] (sawmill) -- node[anchor=west] {includes} (sawing);
\draw [arrow] (sawmill) -- node[anchor=west] {includes} (planing);

\draw [arrow] (sawing) -- node[anchor=west] {produces} (waste);
\draw [arrow] (planing) -- node[anchor=west] {produces} (waste);
\draw [arrow] (waste) -- node[anchor=west] {generates} (emissions);
\draw [arrow] (emissions) -- node[anchor=east] {contains} (fine);
\draw [arrow] (emissions) -- node[anchor=west] {contains} (coarse);
\draw [arrow] (emissions) -- node[anchor=south] {affects} (aqi);
\draw [arrow] (control) -- node[anchor=south] {reduces} (emissions);
\draw [arrow] (emissions) -- node[anchor=west] {tracked by} (profile);
